punish_daimyo_action_BakufuM = {

	category = influence

	require_acceptance = no # Whether the recipient gets an option to decline
	
	is_visible = {
		OR = {
			has_reform = shogunate
			AND = {
				tag = JAP
				any_owned_province = { has_province_modifier = heian_kyo }
			}
		}
		OR = {
			FROM = {
				is_subject_of = ROOT
				is_subject_of_type = daimyo_vassal
			}
			#FROM = { has_reform = indep_daimyo }
		}
		OR = {
			ai = no
			AND = {
				manpower_percentage = 0.10
				ai = yes
			}
		}
	}
	is_allowed = {
		hidden_trigger = {
			OR = {
				ai = no
				AND = {
					ai = yes
					FROM = { NOT = { has_country_modifier = punish_daimyo_action_delay } } #ai cooltime
				}
			}
			NOT = {
				calc_true_if = {
					all_country = { has_country_flag = cb_punish_daimyo_non_target }
					amount = 1
				}
			}
		}
		is_at_war = no
		NOT = { truce_with = FROM }
		has_casus_belli = {
			type = cb_punish_daimyo
			target = FROM
		}
	}

	on_accept = {
		every_country = {
			limit = {
				alliance_with = FROM
				is_subject_of = ROOT
				OR = {
					has_reform = daimyo
					is_subject_of_type = daimyo_vassal
				}
			}
			#grant_independence = yes
			set_country_flag = cb_punish_daimyo_target
		}
		if = {
			limit = { FROM = { is_subject_of = ROOT } }
			FROM = {
				#grant_independence = yes
				set_country_flag = cb_punish_daimyo_target2
			}
		}
		every_country = {
			limit = {
				is_subject_of = ROOT
				OR = {
					has_reform = daimyo
					is_subject_of_type = daimyo_vassal
				}
				NOT = {
					has_country_flag = cb_punish_daimyo_target
					has_country_flag = cb_punish_daimyo_target2
				}
			}
			set_country_flag = cb_punish_daimyo_non_target
		}
		if = {
			limit = { ai = no }
			every_country = {
				limit = {
					OR = {
						has_country_flag = cb_punish_daimyo_target
						has_country_flag = cb_punish_daimyo_target2
					}
				}
				grant_independence = yes
			}
			declare_war_with_cb = {
				who = FROM
				casus_belli = cb_punish_daimyo
			}
			every_country = {
				limit = {
					ai = yes
					is_subject_of = ROOT
					OR = {
						has_reform = daimyo
						AND = {
							is_subject_of = ROOT
							is_subject_of_type = daimyo_vassal
						}
					}
					war_with = FROM
					liberty_desire = 50
				}
				white_peace = FROM
				add_prestige = 25
			}
			every_country = {
				limit = {
					ai = no
					is_subject_of = ROOT
					OR = {
						has_reform = daimyo
						is_subject_of_type = daimyo_vassal
					}
					war_with = FROM
					liberty_desire = 50
				}
				country_event = { id = BakufuM_events.11 }
			}
			#FROM = { country_event = { id = BakufuM_events.7 } }
			every_country = {
				limit = {
					is_subject_of = ROOT
					OR = {
						war_with = FROM
						ai = no
					}
					OR = {
						has_reform = daimyo
						is_subject_of_type = daimyo_vassal
						has_country_flag = bakufuM_DAIMYO_FLAG
					}
				}
				add_liberty_desire = 20
			}
			else = { country_event = { id = BakufuM_events.10 } } #to judge the military difference between shogun and daimyos (for ai)
		}
	}

	ai_will_do = {
		OR = {
			FROM = {
				ai = no
				ROOT = {
					military_strength = {
						who = FROM
						value = 1.5
					}
				}
			}
			military_strength = {
				who = FROM
				value = 1.1
			}
		}
	}

}
