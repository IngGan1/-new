# The Fall of Hungary
country_event = {
	id = austria_hungary.24
	title = austria_hungary.24.t
	desc = austria_hungary.24.d
	picture = BORDER_TENSION_eventPicture

	fire_only_once = yes
	major = yes
	trigger = {
		tag = HUN
		exists = HAB
		exists = TRA
		TRA = {
			is_subject_of = TUR
		}
		is_subject = no
		government = monarchy
		NOT = {
			num_of_cities = 25
		}
		NOT = {
			has_ruler = "Ladislaus Postumus"
		}
		TUR = {
			any_owned_province = {
				is_core = HUN
			}
			owns = 153
			num_of_cities = HAB
		}
		HAB = {
			num_of_cities = HUN
		}
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.5
			dynasty = "von Habsburg"
		}
	}

	option = {
		name = austria_hungary.24.a
		ai_chance = {
			factor = 90
		}
		HAB = {
			create_union = HUN
		}
	}

	option = {
		name = austria_hungary.24.b
		ai_chance = {
			factor = 10
		}
		add_stability = -1
		every_owned_province = {
			add_permanent_claim = HAB
		}
	}
}

# Spanish Acquisition of the Low Countries
country_event = {
	id = austria_hungary.9
	title = spanish_low_countries.t
	desc = spanish_low_countries.d
	picture = GOOD_WITH_MONARCH_eventPicture

	is_triggered_only = yes

	option = { # Let us cede the lands to Spain
		ai_chance = {factor = 100}
		name = spanish_low_countries.a
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						region = low_countries_region
						area = picardy_area
					}
				}
				remove_core = HAB
				add_core = SPA
				cede_province = SPA
			}
			every_subject_country = {
				every_owned_province = {
					limit = {
						OR = {
							region = low_countries_region
							area = picardy_area
						}
					}
					remove_core = HAB
					add_core = SPA
					cede_province = SPA
				}
			}
		}
		custom_tooltip = austria_hungary.9.a.tooltip
		SPA = {
			add_opinion = {
				who = ROOT
				modifier = ceded_netherlands
			}
			country_event = {
				id = austria_hungary.11
			}	
		}
		set_global_flag = low_countries_given_spa
	}

	option = { # Let us keep the lands for ourselves
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
					has_global_flag = governments_extended_historical_yes
			}
		}
		name = spanish_low_countries.b
		SPA = {
			add_opinion = {
				who = ROOT
				modifier = did_not_cede_netherlands
			}
			country_event = {
			id = austria_hungary.12
			}
		}
		every_owned_province = {
			limit = {
				region = low_countries_region
			}
			add_province_modifier = {
				name = wrongful_rulers
				duration = 3650
			}
		}
	}
}
