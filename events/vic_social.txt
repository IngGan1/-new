country_event = {
	id = vic_social.1
	title = "vic_social.1.t"
	desc = "vic_social.1.d"
	picture = REFORM_eventPicture

	trigger = {
		current_age = industrial_age
		NOT = {
			has_country_flag = social_reforms_activated
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	immediate = {
		hidden_effect = {
			set_country_flag = social_reforms_activated
			set_variable = {
 				which = social_min_wage
  				value = 0
			}
			set_variable = {
 				which = social_max_workhours
  				value = 0
			}
			set_variable = {
 				which = social_unemployment
  				value = 0
			}
			set_variable = {
 				which = social_safety
  				value = 0
			}
			set_variable = {
 				which = social_pension
  				value = 0
			}
			set_variable = {
 				which = social_health
  				value = 0
			}
		}
	}

	option = {
		name = "vic_social.1.a"
		trigger = { NOT = { tag = SWE} }
		set_country_flag = vic_social_reform_decision
	}
	option = {
		name = "vic_social.1.b"
		trigger = { tag = SWE }
		set_country_flag = vic_social_reform_decision
	}		
}

country_event = {
	id = vic_social.2
	title = "vic_social.2.t"
	desc = "vic_social.2.d"
	picture = REFORM_eventPicture

	trigger = {
		has_country_flag = social_reforms_activated
		OR = {
			any_neighbor_country = {
				check_variable = {
    				which = social_min_wage
    				value = 1
				}
			}	
			any_rival_country = {
				check_variable = {
    				which = social_min_wage
    				value = 1
    			}	
			}
			any_subject_country = {
				check_variable = {
    				which = social_min_wage
    				value = 1
    			}	
			}
		}
		NOT = {
			OR = {
				any_neighbor_country = {
    				check_variable = {
        				which = social_min_wage
        				value = 2
    				}
				}
				any_rival_country = {
    				check_variable = {
        				which = social_min_wage
        				value = 2
    				}
				}
				any_subject_country = {
					check_variable = {
    					which = social_min_wage
    					value = 2
    				}	
				}
			}
		}	
		NOT = {
    		check_variable = {
        		which = social_min_wage
        		value = 1
    		}
		}
		NOT = {
			has_country_flag = social_min_wage_10	
		}
			
	}

	mean_time_to_happen = {
		months = 120
	}

	immediate = {
		hidden_effect = {
			set_country_flag = social_min_wage_10
			clr_country_flag = social_min_wage_20
			clr_country_flag = social_min_wage_00
		}
	}

	option = {
		name = "vic_social.2.a"
		add_country_modifier = {
			name = social_min_wage_1
			duration = -1
		}
		hidden_effect = {
			remove_country_modifier = social_min_wage_2
		}
	}
}
