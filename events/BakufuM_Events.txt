#check daimyos' cities for Shogunate
country_event = {
	id = BakufuM_events.1
	title = "bakufuM_events.1.t"
	desc = "bakufuM_events.1.d"
	picture = DIPLOMACY_eventPicture

	hidden = yes

	trigger = {
		has_reform = shogunate
		has_country_modifier = FEUDAL_BAKUFU_bakufuM
		OR = {
			NOT = { has_global_flag = CHECK_DAIYOS_CITIES_FLAG }
			variable_arithmetic_trigger = {
				custom_tooltip = bakufu_daimyo_rate_tooltip
				export_to_variable = {
					variable_name = Temp_BakufuCities_Value
					value = num_of_cities
					who = ROOT
				}
				NOT = {
					is_variable_equal = {
						which = Now_BakufuCities_Value
						which = Temp_BakufuCities_Value
					}
				}
			}
			any_province = {
				region = japan_region
				#culture_group = japanese_g
				variable_arithmetic_trigger = {
					custom_tooltip = daimyo_cities_tooltip
					export_to_variable = {
						variable_name = Temp_DaimyoCities_Value
						value = culture
					}
					NOT = {
						is_variable_equal = {
							which = Now_DaimyoCities_Value
							which = Temp_DaimyoCities_Value
						}
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	immediate = {
		hidden_effect = {
			set_global_flag = CHECK_DAIYOS_CITIES_FLAG
			export_to_variable = {
				variable_name = Temp_BakufuCities_Value
				value = num_of_cities
				who = ROOT
			}
			set_variable = {
				which = Now_BakufuCities_Value
				which = Temp_BakufuCities_Value
			}
			every_province = {
				limit = {
					region =japan_region
					#culture_group = japanese_g
					variable_arithmetic_trigger = {
						custom_tooltip = daimyo_cities_tooltip
						export_to_variable = {
							variable_name = Temp_DaimyoCities_Value
							value = culture
						}
						NOT = {
							is_variable_equal = {
								which = Now_DaimyoCities_Value
								which = Temp_DaimyoCities_Value
							}
						}
					}
				}
				export_to_variable = {
					variable_name = Now_DaimyoCities_Value
					value = culture
				}
			}

			set_variable = { 
				which = DaimyosCities_Value
				value = 0
			}
			set_variable = { 
				which = ShogunateCities_InJapan_Value
				value = 0
			}
			set_variable = {
				which = Shogun_Daimyo_Total_Value
				value = 0
			}
			set_variable = {
				which = Shogun_Daimyo_Divided_Value
				value = 0
			}
			every_province = {
				limit = {
					owner = { has_reform = shogunate }
					region = japan_region
				}
				ROOT ={
					change_variable = {
						which = ShogunateCities_InJapan_Value
						value = 1
					}
				}
			}
			every_province = {
				limit = {
					region = japan_region
					culture_group = japanese_g
				}
				ROOT ={
					change_variable = {
						which = DaimyosCities_Value
						value = 1
					}
				}
			}
			change_variable = {
				which = Shogun_Daimyo_Total_Value
				which = DaimyosCities_Value
			}
			change_variable = {
				which = Shogun_Daimyo_Total_Value
				which = ShogunateCities_InJapan_Value
			}
			set_variable = {
				which = Shogun_Daimyo_Divided_Value
				which = ShogunateCities_InJapan_Value
			}
			divide_variable = {            
				which = Shogun_Daimyo_Divided_Value
				which = Shogun_Daimyo_Total_Value
			}
			multiply_variable = {            
				which = Shogun_Daimyo_Divided_Value
				value = 100
			}
		}
	}

	option = {
		name = "BakufuM_events.1.a"
	}
}

				NOT = { legitimacy = 50 }

				NOT = { legitimacy = 60 }
				heir_claim = 60
			}
			AND ={
				NOT = { legitimacy = 70 }
				heir_claim = 70
			}
			AND ={
				NOT = { legitimacy = 80 }
				heir_claim = 80
			}
			AND ={
				NOT = { legitimacy = 90 }
				heir_claim = 90
			}

			AND ={
				legitimacy = 10

			}

				legitimacy = 20

			}
			AND ={
				legitimacy = 30
				NOT = { heir_claim = 30 }
			}
			AND ={
				legitimacy = 40
				NOT = { heir_claim = 40 }
			}
			AND ={
				legitimacy = 50
				NOT = { heir_claim = 50 }
			}
			AND ={
				legitimacy = 60
				NOT = { heir_claim = 60 }
			}
			AND ={
				legitimacy = 70
				NOT = { heir_claim = 70 }
			}
			AND ={
				legitimacy = 80
				NOT = { heir_claim = 80 }
			}
			AND ={
				legitimacy = 90
				NOT = { heir_claim = 90 }
			}

		}
	}

	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = "BakufuM_events.4.a"
		hidden_effect = { set_country_flag = heir_claim_alert }
		add_heir_claim = -100
		if = {
			limit = { NOT = { legitimacy = 10 } }
			add_heir_claim = 5
		}
		if = {
			limit = { legitimacy = 10 NOT = { legitimacy = 20 } }
			add_heir_claim = 15
		}
		if = {
			limit = { legitimacy = 20 NOT = { legitimacy = 30 } }
			add_heir_claim = 25
		}
		if = {
			limit = { legitimacy = 30 NOT = { legitimacy = 40 } }
			add_heir_claim = 35
		}
		if = {
			limit = { legitimacy = 40 NOT = { legitimacy = 50 } }
			add_heir_claim = 45
		}
		if = {
			limit = { legitimacy = 50 NOT = { legitimacy = 60 } }
			add_heir_claim = 55
		}
		if = {
			limit = { legitimacy = 60 NOT = { legitimacy = 70 } }
			add_heir_claim = 65
		}
		if = {
			limit = { legitimacy = 70 NOT = { legitimacy = 80 } }
			add_heir_claim = 75
		}
		if = {
			limit = { legitimacy = 80 NOT = { legitimacy = 90 } }
			add_heir_claim = 85
		}
		if = {
			limit = { legitimacy = 90 NOT = { legitimacy = 100 } }
			add_heir_claim = 95
		}
		if = {
			limit = { legitimacy = 100 }
			add_heir_claim = 100
		}
	}
}

country_event = {
	id = BakufuM_events.7
	title = "bakufuM_events.7.t"
	desc = "bakufuM_events.7.d"
	picture = DIPLOMACY_eventPicture

	trigger = {
		OR = {
			has_country_flag = cb_punish_daimyo_target
			has_country_flag = cb_punish_daimyo_target2
		}
		any_country = {
			is_subject = no
			OR = {
				has_reform = shogunate
				AND = {
					tag = JAP
					any_owned_province = { has_province_modifier = heian_kyo }
				}
			}
			NOT = { war_with = ROOT }
			NOT = { is_subject_of = ROOT }
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	immediate = {
		hidden_effect = {
			clr_country_flag = cb_punish_daimyo_target
			clr_country_flag = cb_punish_daimyo_target2
		}
	}

	option = {
		name = "BakufuM_events.7.a"
		if = {
			limit = { NOT = { has_reform = shogunate } }
			random_country = {
				limit = { has_reform = shogunate }
				create_subject = {
					subject_type = daimyo_vassal
					subject = ROOT
				}
			}
			hidden_effect = { add_legitimacy = 25 }
			else = {
				add_prestige = 25
				hidden_effect = {
					every_country = {
						limit = {
							is_subject_of = ROOT
							has_reform = daimyo
						}
						add_liberty_desire = 20
					}
				}
				every_owned_province = {
					limit = {
						is_core = ROOT
						region = japan_region
					}
					add_nationalism = -20
				}
			}
		}
	}
}

#Punish Daimyo Action for the AI
country_event = {
	id = BakufuM_events.10
	title = "bakufuM_events.10.t"
	desc = "bakufuM_events.10.d"
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes

	immediate = {
		export_to_variable = {
			variable_name = THE_SHOGUN_ARMY_SIZE_VALUE
			value = army_size
		}
		every_country = {
			limit = { has_country_flag = cb_punish_daimyo_target2 }
			export_to_variable = {
				variable_name = THE_DAIMYOS_ARMY_SIZE_VALUE
				value = army_size
			}
		}
		every_country = {
			limit = {
				is_subject_of = ROOT
				NOT = { liberty_desire = 50 }
				NOT = { has_country_flag = cb_punish_daimyo_target }
				NOT = { has_country_flag = cb_punish_daimyo_target2 }
			}
			export_to_variable = {
				variable_name = THE_SHOGUN_ARMY_SIZE_VALUE_2
				value = army_size
			}
			while = {
				limit = {
					check_variable = {
						which = THE_SHOGUN_ARMY_SIZE_VALUE_2
						value = 1
					}
				}
				change_variable = {            
					which = THE_SHOGUN_ARMY_SIZE_VALUE_2
					value = -1
				}
				ROOT = {
					change_variable = {            
						which = THE_SHOGUN_ARMY_SIZE_VALUE
						value = 1
